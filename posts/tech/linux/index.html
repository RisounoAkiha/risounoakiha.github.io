系统架构

 1#系统信息
 2uname -a
 3#登录提示或系统标识
 4cat /etc/issue
 5#被解析获取信息的描述行
 6cat /etc/*-release //debien
 7cat /etc/redhat-release //red hat
 8#内核版本，gcc版本，内核编译时间，用户名
 9cat /proc/version
10#调整监视linux内核操作各种活动的文件
11cat /proc/sys/kernel/version

进程

 1#列出当前进程快照
 2ps -elf
 3ps aux
 4#杀死进程
 5kill -9 pid
 6#运行时的需要信息（proc是虚拟文件系统称为过程信息的伪文件系统）
 7ls -al /proc/
 8#关于对应PID99的进程的信息
 9ls -al /proc/99
10#进程实时显示
11top

 1#用户的UID和GID
 2id

 4w
 5#当前用户名
 6whoami

 8last

10lastb
11#打印用户登录日志/var/log/lastlog文件的内容(var目录有各种软件的日志)
12lastlog

14cat /etc/passwd
15#root访问，散列密码数据
16cat /etc/shadow
17#sudo命令需要遵守的规则
18cat /etc/sudoers
19#sudo版本字符串
20sudo -V
21#列出目前用户可执行与无法执行的命令
22sudo -l

SSH服务信息

私钥需要权限600。最好不要开密码登录，防止ssh爆破。

 1#SSH登录
 2ssh 用户名@ip
 3#查看公钥认证
 4cat ~/.ssh/authorized_keys
 5#公钥
 6Cat ~/.ssh/id_rsa.pub
 7#私钥
 8cat ~/.ssh/id_rsa
 9#OpenSSH控制端配置文件
10cat /etc/ssh/ssh_config
11#OpenSSH服务端配置文件
12cat /etc/ssh/sshd_config

 2service --status-all
 3#列出安装在文件系统中的所有单元
 4systemctl -a
 5#启动某个服务
 6service servicename start
 7systemctl start servicename
 8#停止某个服务
 9service servicename stop
10systemctl stop servicename
11#显示某个服务状态信息
12service servicename status
13systemctl status servicename
14#查看端口号对应的服务
15cat /etc/services

安全

 1#列出所有规则链
 2iptables -L
 3#删除选定规则链的所有规则
 4tptables -F
 5#A指定规则，p是协议，m模板，j是行为。下面这个是一条规则，具体学习参考文档。
 6iptables -A INPUT -p icmp -m icmp -icmp-type 0 -j ACCEPT
 7#允许来自str端口80的tcp连接
 8iptables -A INPUT -p tcp -m tcp -sport 80 -m -state RELATED,ESTABLISHED -j ACCEPT
 9#允许从/到dst端口80的tcp连接
10iptables -A OUTPUT -p tcp -m tcp -dport 80 -m -state NEW,RELATED,ESTABLISHED -j ACCEPT
11#允许来自src端口80的udp连接
12iptables -A INPUT -p udp -m udp -sport 53 -m -state RELATED,ESTABLISHED -j ACCEPT

文件系统

 1#linux系统环境和启动程序
 2cat /etc/profile
 3#shell的启动文件
 4cat /etc/bashrc
 5#适用于当前用户的bashrc
 6cat ~/.bash_profile
 7#bash命令的历史记录
 8cat ~/.bash_history
 9#立刻写入history
10history -w 
11#所有日志文件
12ls -al /var/log
13#找粘滞键（只有目录所有者，文件所有者可以在这里删除或重命名）
14#/是所有目录，perm是权限，type是类型，d是目录，0是stdin，1是stdout，2是标准报错，2>/dev/null是将报错信息忽略，/dev/null是特殊文件，不输出的意思。
15find / -perm -100 -type d 2>/dev/null
16#SGID（作为组运行，而不是启动它的用户）
17find / -perm -g=s -type f 2>/dev/null
18#SUID（作为所有者运行，而不是启动它的用户）
19find / -perm -u=s -type f 2>/dev/null
20#找出可写的文件夹
21find / -writable -type d 2>/dev/null
22find / -perm -222 -type d 2>/dev/null
23find / -perm -o w -type d 2>/dev/null
24#可执行的文件夹
25find / -perm -o x -type d 2>/dev/null
26#可写可执行
27find / (-perm -o x -perm -o w) -type d 2>/dev/null
28#可写
29find / -xdev -type d (-perm -0002 -a ! -perm -1000) -print
30#找不是所有者的文件
31find /dir -xdev (-nouser -o -nogroup) -print

程序

 2crontab -l
 3cat /etc/contab

 5ls -alh /var/spool/cron

 7cat /etc/cron*
 8#列出所有用户的crontab文件
 9ls -al /var/spool/cron/crontabs
10#打印root用户的crontab
11car /var/spool/cron/crontabs/root
